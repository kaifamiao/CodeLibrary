//quick sort
int cmp( const void * a , const void * b ){

    return *( int * )a - *( int * )b;

}

int * getLeastNumbers( int * arr , int arrSize , int k , int * returnSize ){

    *returnSize = k;

    qsort( arr , arrSize , sizeof( int ) , cmp );

    return arr;

}
