func quickSort(nums []int) {
	ln := len(nums)
	if ln < 2 {
		return
	}

	var (
		tag        = nums[0]
		head, tail = 0, ln - 1
	)

	for i := 1; i <= tail; {
		if nums[i] < tag {
			nums[head], nums[i] = nums[i], nums[head]
			head++
			i++

			continue
		}

		nums[tail], nums[i] = nums[i], nums[tail]
		tail--
	}

	quickSort(nums[:head])
	quickSort(nums[head+1:])
}
