class Solution {
    public int firstMissingPositive(int[] nums) {
        Arrays.sort(nums);
        int NOtShowMin=1;
        int len=nums.length;
        for(int i=0;i<len;i++){
            if(nums[i]==NOtShowMin){
                NOtShowMin++;
            }
        }
        return NOtShowMin;
    }
}
