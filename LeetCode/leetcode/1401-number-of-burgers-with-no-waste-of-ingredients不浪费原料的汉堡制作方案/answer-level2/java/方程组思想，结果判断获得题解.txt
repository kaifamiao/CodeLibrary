public List<Integer> numOfBurgers(int tomatoSlices, int cheeseSlices) {

        List<Integer> list = new ArrayList<>();
        if(tomatoSlices % 2 == 0)
        {
            int x = 2 * cheeseSlices - tomatoSlices / 2;
            int y = tomatoSlices / 2 - cheeseSlices;
            if (x % 1 == 0 && x >= 0 && y % 1 == 0 && y >= 0)
            {
                list.add(y);
                list.add(x);
            }
        }
        return list;

    }
