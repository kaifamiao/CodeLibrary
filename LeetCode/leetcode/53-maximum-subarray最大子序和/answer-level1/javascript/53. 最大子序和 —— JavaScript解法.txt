var maxSubArray = function(nums) {
    const dp = [nums[0]], len = nums.length
    let max = dp[0]
    for (let i = 1; i < len; i++) {
        dp[i] = Math.max(dp[i - 1] + nums[i], nums[i])
        max = Math.max(dp[i], max)
    }
    return max
};
