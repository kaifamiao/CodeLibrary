char * lastSubstring(char * s){
    int len = strlen(s);
    int ans = 0;
    for (int i = 1; i < len; i++) {
        if (s[i] <= s[i - 1]) {
            continue;
        }
        if (strcmp(&s[i], &s[ans]) > 0) {
            ans = i;
        }
    }
    return &s[ans];
}
