struct ListNode* removeElements(struct ListNode* head, int val){
    struct ListNode * newpoint, *p, *lastpoint;
    if(head==NULL)return head;
    while(head->val==val){
        if(head->next!=NULL)head=head->next;
        else return NULL;
    }
    p=head;
    if(head->next!=NULL){
        lastpoint=p;
        p=p->next;
    }
    while(p->next!=NULL){
        if(p->val==val){
            p=p->next;
            lastpoint->next=p;
        }
        else{
            p=p->next;
            lastpoint=lastpoint->next;
        }
    }
    if(p->val==val){
        lastpoint->next=NULL;
    }
    return head;
}
