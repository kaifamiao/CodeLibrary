```python []
class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        d = collections.Counter(i for i, j in zip(s1, s2) if i != j)
        if (d['x'] + d['y']) & 1 or len(s1) != len(s2):
            return -1
        return sum(divmod(d['x'], 2) + divmod(d['y'], 2))
```
