func numWays(n int) int {
    if n == 0 { 
        return 1
    }
    if n == 1 {
        return 1
    }
    if n == 2 {
        return 2
    }
    pre, curr := 1, 2
    for i := 3; i <= n; i++ {
        pre, curr = curr, (pre + curr) % 1000000007
    }
    return curr
}
