var reverseKGroup = function(head, k) {
  let newHead = new ListNode( null ); // è¿”å›çš„æ–°é“¾è¡¨
  newHead.next = head;
  let p = newHead; // æ–°é“¾è¡¨çš„æŒ‡é’ˆ
  let prev = null,

      next = null,
      count = k;
  while (curr !== null) {
    count = k;
    
    let headcount = 0, temp = curr;
    while (temp !== null) {
      temp = temp.next;
      headcount++;
    }
    if (headcount < k) break;
    
    while (count-- > 0 && curr !== null) {
      next = curr.next;
      curr.next = prev;
      prev = curr;

    p.next = prev;
    p = pnext;
    p.next = curr;
  }
  
  return newHead.next;
}
