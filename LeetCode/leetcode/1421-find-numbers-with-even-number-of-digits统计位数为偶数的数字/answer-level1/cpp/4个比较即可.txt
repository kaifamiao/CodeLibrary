class Solution {
public:
    int findNumbers(vector<int>& nums) {
        int count=0;
        int x;
        for(int i=0;i<nums.size();i++){
        	x=nums[i];
        	if((x>=10&&x<=99)||(x>=1000&&x<=9999))
        		count++;
		}
		return count;
    }
};
