/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public boolean hasCycle(ListNode head) {
        if(head==null||head.next==null){return false;}
        ListNode ln = head;
        ListNode stop = new ListNode(0);
        ListNode temp;
        while(ln!=null){
            if(ln==stop){
                return true;
            }
            temp=ln;
            ln=ln.next;
            temp.next=stop;
        }
        return false;
    }
}
