class Solution:
    def hasCycle(self, head: ListNode) -> bool:
        dic = {}
        while head:
            if head in dic:
                return True
            else:
                dic[head] = 1
            head = head.next
        return False
```

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def hasCycle(self, head: ListNode) -> bool:
        
        if not head:
            return False
        temp1 = head.next
        if not temp1:
            return False
        temp2 = temp1.next
        
        while head and temp2:
            if head == temp2:
                return True
            head = head.next
            if temp2.next:
                temp2 = temp2.next.next
            else:
                return False
        return False
```
