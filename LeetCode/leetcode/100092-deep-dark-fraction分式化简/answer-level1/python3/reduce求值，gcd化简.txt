```
from math import gcd
from functools import reduce
class Solution:
    def fraction(self, cont: List[int]) -> List[int]:
        res=reduce(lambda a,b:[b*a[0]+a[1],a[0]],cont[::-1],[1,0])
        return [i//gcd(*res) for i in res]
```
