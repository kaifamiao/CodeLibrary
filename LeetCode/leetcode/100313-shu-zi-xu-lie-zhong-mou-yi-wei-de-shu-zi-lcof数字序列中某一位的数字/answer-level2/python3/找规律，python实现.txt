class Solution:
    def findNthDigit(self, n: int) -> int:
        if n+1 <= 10: return n
        tim = 1
        res = 0
        dif = tim * 9 * 10 ** (tim - 1)
        while n > dif:
            n -= dif
            res += 9 * 10 ** (tim - 1)
            tim += 1
            dif = tim * 9 * 10 ** (tim - 1)
        res += n//tim
        tmp = n % tim
        if tmp:
            res += 1
            return int(str(res)[tmp-1])
        else:
            return int(str(res)[-1])
