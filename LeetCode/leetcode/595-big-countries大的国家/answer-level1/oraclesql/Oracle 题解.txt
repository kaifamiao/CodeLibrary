方法一：or
select w.name,w.population,w.area 
from world w 
where w.area >3000000 or w.population >25000000

方法二：union
select w.name,w.population,w.area
from world w
where w.area>3000000
union
select w.name,w.population,w.area
from world w
where w.population>25000000
