class Solution {
    public int[] productExceptSelf(int[] nums) {
        int[] sum=new int[nums.length];
        int t=1;
        for(int i=0;i<nums.length;i++){
            sum[i]=t;
            t*=nums[i];
        }
        t=1;
        for(int j=(sum.length-1);j>=0;j--){
            sum[j]*=t;
            t*=nums[j];
        }
        return sum;
    }
