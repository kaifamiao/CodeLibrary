class Solution:
    def removeKdigits(self, num: str, k: int) -> str:
        num_length = len(num)
        if num_length <= k:
            return "0"
        
        left_count = num_length - k
        res = ""
        count = 0
        begin = 0
        
        while count < left_count:
            min_index = 0
            min_num = float('inf')
            for i in range(begin, k + count + 1):
                # æ‰¾åˆ°æœ€å°æ•°
                if int(num[i]) < min_num:
                    min_index = i
                    min_num = int(num[i])
            # æŠŠæ‰¾åˆ°çš„æœ€å°æ•°åŠ å…¥ç»“æžœåˆ—è¡¨
            res += str(min_num)
