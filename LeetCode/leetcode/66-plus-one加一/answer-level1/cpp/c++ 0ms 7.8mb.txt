class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {
        bool flag=1;
        for(int i=digits.size()-1;i>=0;--i){
            if(flag+digits[i]>=10){
                digits[i]=(flag+digits[i])%10;
                flag=1;
            }
            else {
                    digits[i]=(flag+digits[i]);
                    flag=0;
            }
        }
        if(flag) { digits.insert(digits.begin(),1);return digits;}
        return digits; 
    }
};
