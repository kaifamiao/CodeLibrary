class Solution(object):
    def plusOne(self, digits):
        """
        :type digits: List[int]
        :rtype: List[int]
        """
        ien = len(digits)-1
        while ien>=0:
            if digits[ien]!=9:
                digits[ien]+=1
                return digits
            digits[ien]=0
            ien-=1
        digits.insert(0,'1')
        return digits
