```
class Solution:
    def plusOne(self, digits: List[int]) -> List[int]:
        from functools import reduce
        ## 数字转字符串，字符串拼接，拼接后识别成数字，加一，再转为字符串，再拆分
        return list(str(eval(reduce(lambda x,y:x+y,map(str,digits)))+1)) 
```
