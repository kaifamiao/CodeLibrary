class Solution(object):
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        if len(nums)<2:
            return [nums]
        res = []
        for i in range(len(nums)):
            ans = [[nums[i]]+j for j in self.permute([num for num in nums if num != nums[i]])]
            res +=ans
        return res
