class Solution {
public:
    string reverseWords(string s) {
        if (s.size() <= 1) return s;

        auto it1 = s.begin(), it2 = it1;
        while (it2 != s.end()) {
            while (it2 != s.end() && *it2 != ' ') {
                ++it2;
            }
            std::reverse(it1, it2);
            while (it2 != s.end() && *it2 == ' ') {
                ++it2;
            }
            it1 = it2;
        }

        return s;
    }
};
