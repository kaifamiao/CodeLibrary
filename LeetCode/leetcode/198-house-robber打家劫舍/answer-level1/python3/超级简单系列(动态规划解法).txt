def rob(nums):

    m = len(nums)

    if not m:
        return 0

    dp = [0] * (m + 1)

    dp[1] = nums[0]

    for i in range(1,m):

        dp[i+1] = max(dp[i], dp[i-1]+nums[i])

    print(dp)
    return dp[-1]
