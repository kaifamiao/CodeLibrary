/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function(prices) {
  if (prices.length === 0) {
    return 0
  }
  let res = 0
  let min = prices[0]

  for (const price of prices) {
    if (price < min) {
      min = price
      continue
    }

    if (res < price - min) {
      res = price - min
      continue
    }
  }

  return res
}
