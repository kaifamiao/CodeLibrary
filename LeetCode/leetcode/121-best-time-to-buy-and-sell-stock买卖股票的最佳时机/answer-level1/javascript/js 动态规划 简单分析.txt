### 解题思路
```js
  分析状态：
  每天有两种情况：
  1. 手里持有股票
     - 昨天就有，今天没卖
     - 今天刚卖出
  2. 手里没有股票
     - 今天刚买的
     - 昨天就没有
```

### 代码

```javascript
/**
 * @param {number[]} prices
 * @return {number}
 */

var maxProfit = function(prices) {
  let dp = [];
  for (let i = 0; i < prices.length; i++) dp[i] = [];
  
  dp[-1] = [];
  dp[-1][0] = 0;
  dp[-1][1] = -prices[0];
  
  for (let i = 0; i < prices.length; i++) {
    dp[i][0] = Math.max(dp[i - 1][0], dp[i - 1][1] + prices[i]);
    dp[i][1] = Math.max(dp[i - 1][1], -prices[i]);
  }
  
  return dp[prices.length - 1][0];
}
```
