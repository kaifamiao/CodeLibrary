var longestCommonPrefix = function(strs) {
        if(strs.length ==0){
            return ""
        }else if(strs.length == 1){
            return strs[0]
        }else{
            let firstStrs = strs[0]
            let result = ''
            for(var i=0;i<firstStrs.length;i++){
                for(var j=1;j<strs.length;j++){
                    if(firstStrs[i]!=strs[j][i])
                        return result
                }
                result+=firstStrs[i];
            }
            return result;
        }
    };
