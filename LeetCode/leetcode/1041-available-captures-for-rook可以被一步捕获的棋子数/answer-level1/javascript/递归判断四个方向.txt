var numRookCaptures = function(board) {
    let [x, y, sum] = [0, 0, 0];
    let arr = [];
    //æ‰¾åˆ°è½¦æ‰€åœ¨çš„åæ ‡
    for(let i = 0; i < 8; i ++) {
        if(board[i].indexOf('R') !== -1) {
            x = i;
            y = board[i].indexOf('R');
            break;
        }
    }
    //é€’å½’è°ƒç”¨  ç¢°åˆ°è¾¹ç•Œæˆ–è€…ğŸ˜åˆ™æ¢ä¸€ä¸ªæ–¹å‘ï¼Œé‡åˆ°å°å’num + 1 è°ƒæ•´æ–¹å‘æ»¡å››æ¬¡ä»¥åè¿”å›num
    const find = (a,b,board) => {
        if(arr.length === 4) return sum;
        if(board[a][b] === 'B') {
            arr.push(1);
            return find(x, y, board);
        }
        if(board[a][b] === 'p') {
            sum ++;
            arr.push(1);
            return find(x, y, board)
        }
        if(arr.length === 0 && a - 1 >= -1) {
            if(a - 1 === -1) {
                arr.push(1);
                return find(x, y, board);
            }
            return find(a - 1, b, board);
        }
        if(arr.length === 1 && a + 1 <= 8) {
            if(a + 1 === 8) {
                arr.push(1);
                return find(x, y, board);
            }
            return find(a + 1, b, board);
        }
        if(arr.length === 2 && b - 1 >= -1) {
            if(b - 1 === -1) {
                arr.push(1);
                return find(x, y, board);
            }
            return find(a, b - 1, board);
        }
        if(arr.length === 3 && b + 1 <= 8) {
            if(b + 1 === 8) {
                arr.push(1);
                return find(x, y, board);
            }
            return find(a, b + 1, board);
        }
    }
    return find(x,y,board)
};
