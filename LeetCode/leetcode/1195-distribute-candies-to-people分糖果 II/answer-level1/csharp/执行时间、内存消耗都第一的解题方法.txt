public class Solution {
    public int[] DistributeCandies(int candies, int num_people) {
        int i=0;
        int give=1; //give为这次我要给出的糖果数量
        int[] answer=new int[num_people];
        while(true){
            if(give<candies){
                answer[i]+=give;
                candies-=give;
                give++;
            }
            else if(give>=candies){
                answer[i]+=candies;
                return answer;
            }
            i++;
            if(i==num_people){i=0;}
        }
    }
}
