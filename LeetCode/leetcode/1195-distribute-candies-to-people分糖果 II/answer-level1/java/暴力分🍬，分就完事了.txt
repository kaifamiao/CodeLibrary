class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] xpy = new int[num_people];
        int c = 1;
        int i = 0;
        while(candies > 0) {
            xpy[i % num_people] += candies > c ? c : candies;
            candies -= c;
            c += 1;
            i += 1;
        }
        return xpy;
    }
}
