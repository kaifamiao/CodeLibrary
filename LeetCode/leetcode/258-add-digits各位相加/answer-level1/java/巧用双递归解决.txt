class Solution {
    public int addDigits(int num) {
        if(num < 10){
            return num;
        }else{
            return addDigits(count(num));
        }
    }

    public int count(int n){
        if(n == 0){
            return n;
        }else{
            return n%10 + count(n/10);
        }
    }
}
