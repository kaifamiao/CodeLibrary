状态机的思路解，画出图之后思路比较清晰，代码简单

```
class Solution {
public:
    vector<string> removeComments(vector<string>& source) {
        string cur, status = "str";
        vector<string> res;
        for (string s : source) {
            for (char ch : s) {
                if (status == "str") {
                    if (ch == '/') {
                        status = "pre";
                    }
                    else cur += ch;
                }
                else if (status == "pre") {
                    if (ch == '/') {
                        status = "str";
                        break;
                    }
                    else if (ch == '*') {
                        status = "block";
                    }
                    else {
                        status = "str";
                        cur += '/';
                        cur += ch;
                    }
                }
                else if (status == "block") {
                    if (ch == '*') status = "block_end_pre";
                }
                else if (status == "block_end_pre") {
                    if (ch == '/') status = "str";
                    else if (ch != '*') status = "block";
                }
            }
            if (status == "pre") {
                cur += '/';
                status = "str";
            }
            else if (status == "block_end_pre") status = "block";
            //cout << status << endl;
            if (cur.size() != 0 && status == "str") {
                res.push_back(cur);
                cur = "";
            }
        }
        return res;
    }
};
```
