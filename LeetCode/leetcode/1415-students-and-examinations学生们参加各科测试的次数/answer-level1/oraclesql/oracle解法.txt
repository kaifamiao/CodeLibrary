```
SELECT S.STUDENT_ID,
       S.STUDENT_NAME,
       S.SUBJECT_NAME,
       NVL(T.ATTENDED_EXAMS, 0) ATTENDED_EXAMS
  FROM (SELECT * FROM STUDENTS T, SUBJECTS S) S,
       (SELECT E.STUDENT_ID,
               E.SUBJECT_NAME,
               COUNT(E.SUBJECT_NAME) ATTENDED_EXAMS
          FROM EXAMINATIONS E
         GROUP BY E.STUDENT_ID, E.SUBJECT_NAME) T
 WHERE S.STUDENT_ID = T.STUDENT_ID(+)
   AND S.SUBJECT_NAME = T.SUBJECT_NAME(+)
 ORDER BY S.STUDENT_ID, S.SUBJECT_NAME
```
