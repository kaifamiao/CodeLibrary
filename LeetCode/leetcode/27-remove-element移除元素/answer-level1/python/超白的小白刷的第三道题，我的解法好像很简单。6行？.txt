class Solution(object):
    def removeElement(self, nums, val):
        """
        :type nums: List[int]
        :type val: int
        :rtype: int
        """
        p=0
        while(p<len(nums)):
            if nums[p]！=val:
                p+=1
            else:
                nums.pop(p)
