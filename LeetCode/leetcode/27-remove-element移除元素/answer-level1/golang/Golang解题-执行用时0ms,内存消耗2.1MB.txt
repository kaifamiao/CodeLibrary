func removeElement(nums []int, val int) int {
    if len(nums) == 0 {
        return 0
    }
    
    for i := 0; i < len(nums);{
        if nums[i] == val {
            nums = append(nums[:i], nums[i+1:]...)
            continue
        } else {
            i++
        }
    }
    return len(nums)
}
