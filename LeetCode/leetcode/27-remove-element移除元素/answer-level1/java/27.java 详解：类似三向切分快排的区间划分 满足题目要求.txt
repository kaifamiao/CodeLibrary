public int removeElement(int[] nums, int val) {
        if(nums == null || nums.length == 0)
            return 0;
        int hi = nums.length - 1, lo = 0;
        while(lo <= hi){
            if(nums[lo] == val){
                nums[lo] = nums[hi];
                hi--;
            } else {
                lo++;
            }
        }
        return lo;
    }
