```
class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        int k=0;//定义步数，把后面的元素往前面复制
        for(int i=0;i<nums.size();i++)
        {
            nums[i-k]=nums[i];
            if(nums[i]==val)k++;
        }
        return nums.size()-k;
    }
};
```
