class Solution {
    public int orangesRotting(int[][] grid) {
        if (grid == null) return 0;

        int[][] directions = new int[][]{{-1,0}, {0,-1}, {1,0}, {0,1}};
        Queue<int[]> queue = new ArrayDeque<>();
        int rows = grid.length;
        int cols = grid[0].length;

        // å°†è…çƒ‚å¥å­ğŸŠçš„ä½ç½®å­˜å…¥é˜Ÿåˆ—ï¼Œä¸€æ­¥æ­¥æ„ŸæŸ“
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (grid[i][j] == 2) queue.offer(new int[]{i,j,0});
            }
        }
        // å¼€å§‹æ„ŸæŸ“äº†
        int time = 0;
        while (!queue.isEmpty()){
            int[] curr = queue.poll();
            time = curr[2];
            // å››ä¸ªæ–¹å‘æœç´¢ï¼Œå°†å¥½æ©˜å­ç ´å
            for (int k = 0; k < 4; k++) {
                int newX = curr[0] + directions[k][0];
                int newY = curr[1] + directions[k][1];
                if (newX >= 0 && newX < rows && newY >= 0 && newY < cols && grid[newX][newY] == 1) {
                    grid[newX][newY] = 2;
                    queue.offer(new int[]{newX, newY, time + 1});
                }
            }
        }

        // æœç´¢è¿˜æœ‰æ²¡æœ‰æ–°é²œçš„æ©˜å­ğŸŠ
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (grid[i][j] == 1) return -1;
            }
        }
        return time;   
    }
}
