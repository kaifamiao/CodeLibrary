class Solution {
    public boolean isValidBST(TreeNode root) {
        if(root == null) return true;
        double min = -Double.MAX_VALUE;
        Stack<TreeNode> stack = new Stack<>();
        while(!stack.isEmpty() || root !=  null) {
            if (root != null) {
                stack.push(root);
                root = root.left;
            }else {
                root = stack.pop();
                if (root.val <= min) return false;
                min = root.val;
                root = root.right;
            }
        }
        return true;
    } 
}
