class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        int s1=0,s2=0,s3=0;
        bool succ=false;
        for(int i=0;i<A.size()-2;i++){
            for(int j=0;j<=i;j++)
                s1+=A[j];
            for(int j=i+1;j<A.size()-1;j++){
                s2+=A[j];
                if(s2==s1){
                    for(int k=j+1;k<A.size();k++)
                        s3+=A[k];
                    if(s3==s2)
                        succ=true;
                }
            }
            if(succ==true){
                break;
            }else{
                s1=s2=s3=0;
            }
        }
        return succ;
    }
};
