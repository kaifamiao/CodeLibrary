var findDisappearedNumbers = function(nums) {
    const length = nums.length;
    if (!length) {
        return [];
    }

    const map = {};
    nums.forEach(num => (map[num] = true));

    const res = [];

        if (!map[i]) res.push(i);
    }

    return res;
};
