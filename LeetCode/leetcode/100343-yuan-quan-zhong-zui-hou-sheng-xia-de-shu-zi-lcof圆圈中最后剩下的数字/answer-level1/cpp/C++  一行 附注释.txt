    /*
    f[n][m] : 表示n个人 每数m个删除 最后一个被删的编号是多少 编号从0开始
    若已知f[n-1][m] = x 
    则 对于n-1个人时 此时编号为0的点 即为n个人时第一次被删的点的后一个元素 其编号为 (m-1+1) % n 即为偏移量
    将下一层提取到上一层只需要把n-1层时的最后删除的编号+偏移量即可
    则f[n][m] = (m % n + x) % n = (m + x) % n
    特别的 n=1时 自然就是第一个人了 编号为 0  有f[1][m] = 0
    */
    int lastRemaining(int n, int m) {
        return n==1 ? 0 : (m + lastRemaining(n-1, m)) % n;
    }
