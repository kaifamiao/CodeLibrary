```
class Solution {
public:
    int lastRemaining(int n, int m) {
        int ans = 0;// 最后一轮(第n轮)，重新编号后最后剩余的编号
        for(int i=2;i<=n;i++){// 不断映射上一轮对应的编号
            ans = (ans+m)%i;
        }
        return ans;
    }
};
```
