### 解题思路
套union find的模板。

### 代码

```java
class Solution {
  public int earliestAcq(int[][] logs, int N) {
    Arrays.sort(logs, (a, b) -> {
			return a[0] - b[0];
		});

    Friends f = new Friends(N);

    for (int[] record : logs) {
      f.union(record[1], record[2]);
      if (f.circles == 1) return record[0];
    }

    return -1;
  }
}

class Friends {
  int n;
  int circles;
  int[] leader;

  Friends(int N) {
    n = N;
    circles = N;
    leader = new int[N];

    for (int i = 0; i < N; i++) {
      leader[i] = i;
    }
  }

  void union(int A, int B) {
    int ALeader = find(A);
    int BLeader = find(B);

    if (ALeader != BLeader) {
      leader[ALeader] = BLeader;
      circles--;
    }
  }

  private int find(int A) {
    while (leader[A] != A) {
      A = leader[A];
    }
    return A;
  }
}
```
