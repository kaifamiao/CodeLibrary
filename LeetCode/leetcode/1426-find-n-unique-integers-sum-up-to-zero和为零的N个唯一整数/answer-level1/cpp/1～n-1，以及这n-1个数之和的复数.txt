```
class Solution {
public:
    vector<int> sumZero(int n) {
        vector<int> re;
        int sum=0;
        for(int i=1;i<=n-1;i++){
        	sum+=i;
        	re.push_back(i);
		}
		re.push_back(-1*sum);
		return re;
    }
};
```
