func duplicateZeros(arr []int)  {
	res := make([]int, len(arr))
	copy(res, arr)
	temp := 0
	for _, v := range res{
		if temp >= len(arr) {
			break
		}
		if v == 0 {
			arr[temp] = 0
			if temp + 1 < len(arr) {
				arr[temp + 1] = 0
			}
			temp ++
		} else {
			arr[temp] = v
		}
		if temp + 1 == len(arr) {
			break
		}
		temp ++
	}
}
