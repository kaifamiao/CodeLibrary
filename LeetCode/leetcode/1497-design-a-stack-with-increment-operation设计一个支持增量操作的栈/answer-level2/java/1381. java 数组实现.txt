public class CustomStack {
    private int[] aux;
    private int n;
    public CustomStack(int maxSize) {
        aux = new int[maxSize];
        n = 0;
    }

    public void push(int x) {
        if(n == aux.length)
            return;
        aux[n++] = x;
    }

    public int pop() {
        if(n == 0) return -1;
        return aux[--n];
    }

    public void increment(int k, int val) {
        int boundary = Math.min(k, n);
        for(int i = 0; i < boundary; i++)
            aux[i] += val;
    }
}
