class Solution {
public:
	int jump(vector<int>& nums) {
		vector<int> stepMap;
		stepMap.push_back(0);
		for (int i = 0; i < nums.size(); i++) {
			int step = stepMap[i] + 1;
			int walk = nums[i];
			for (int j = stepMap.size(); j <= i + walk; j++) {
				stepMap.push_back(step);
			}
			if (stepMap.size() == nums.size())
				return stepMap[nums.size() - 1];
		}
		return stepMap[nums.size() - 1];
	}
};
