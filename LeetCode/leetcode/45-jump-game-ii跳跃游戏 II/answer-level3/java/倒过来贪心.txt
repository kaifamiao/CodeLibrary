```
public int jump(int[] nums) {
        int res = 0;
        int cur = nums.length -1;

        while (cur > 0){
            for (int i = 0 ; i < cur ; i++){
                if (nums[i] + i >= cur){
                    cur = i;
                    res++;
                    break;
                }
            }
        }

        return res;
    }
```
