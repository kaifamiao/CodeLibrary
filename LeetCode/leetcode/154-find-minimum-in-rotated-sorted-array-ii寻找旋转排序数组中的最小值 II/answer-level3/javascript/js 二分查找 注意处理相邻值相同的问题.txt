/**
 * @param {number[]} nums
 * @return {number}

var findMin = function(nums) {
  let ans = Infinity,
      low = 0,
      high = nums.length - 1;
  
  while (low <= high) {
    while (nums[low] === nums[low + 1] && low <= high) low++;
    while (nums[high] === nums[high - 1] && high >= low) high--;
    
    let mid = low + ((high - low) >> 1); // ä½è¿ç®—ç¬¦ï¼Œç›¸å½“äº low + Math.floor((high - low) / 2)
    
    if (nums[low] <= nums[mid]) {
      ans = Math.min(ans, nums[low]);
      low = mid + 1;
    } else {
      ans = Math.min(ans, nums[mid]);
      high = mid - 1;
    }
  }
  
  return ans;
};
