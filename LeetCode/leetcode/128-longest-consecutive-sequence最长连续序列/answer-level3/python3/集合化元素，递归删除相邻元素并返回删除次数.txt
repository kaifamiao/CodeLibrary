```
class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        d=set(nums)
        
        def f(i):
            if i in d:
                d.remove(i)
                return 1+f(i+1)+f(i-1)
            return 0
        
        ans=0
        for i in nums:
            ans=max(ans,f(i))
        
        return ans
```
