class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:
        res = []
        top = 0
        bottom = len(matrix) - 1
        if bottom < 0:
            return res
        left = 0
        right = len(matrix[0]) - 1

        while left <= right and top <= bottom:
            j = left
            while j <= right:
                res.append(matrix[top][j])
                j += 1
            top += 1

            i = top
            if i > bottom:
                break
            while i <= bottom:          
                res.append(matrix[i][right])
                i += 1
            right -= 1

            j = right
            if j < left:
                break
            while j >= left:
                res.append(matrix[bottom][j])
                j -= 1
            bottom -= 1

            i = bottom
            if i < top:
                break
            while i >= top:
                res.append(matrix[i][left])
                i -= 1
            left += 1

        return res


# @lc code=end
