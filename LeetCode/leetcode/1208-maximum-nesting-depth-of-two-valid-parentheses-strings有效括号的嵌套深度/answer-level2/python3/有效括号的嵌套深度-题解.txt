class Solution:
    def maxDepthAfterSplit(self, seq: str) -> List[int]:

        nums = len(seq)
        stack = []
        for i in range(nums):
            if seq[i] == '(':
                stack.append(i%2)
            else:
                stack.append((i+1)%2)
        
        return stack
