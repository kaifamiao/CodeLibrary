class Solution(object):
    def diStringMatch(self, S):
        """
        :type S: str
        :rtype: List[int]
        """
        left, right, L = 0, len(S), []
        for i in S:
            if i == 'I':
                L.append(left)
                left += 1
            else:
                L.append(right)
                right -= 1
        # å‰©ä¸‹çš„å€¼æ— è®ºæ˜¯left æˆ– rightéƒ½æ˜¯å¯ä»¥æ»¡è¶³æ¡ä»¶çš„
        return L + [left]
