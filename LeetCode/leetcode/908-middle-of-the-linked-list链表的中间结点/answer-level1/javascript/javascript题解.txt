## 双指针

```javascipt
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var middleNode = function(head) {
    let slow = head,
        fast = head;
    
    while (fast !== null) {
        fast = fast.next;
        if (fast !== null) {
            fast = fast.next;
        } else {
            return slow;
        }
        slow = slow.next;
    }
    return slow;
};
```
