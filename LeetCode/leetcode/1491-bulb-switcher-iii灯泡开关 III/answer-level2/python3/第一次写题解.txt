```
class Solution:
    def numTimesAllBlue(self, light: List[int]) -> int:
        summ, index, ins = 0, 0, 0
        for i in range(len(light)):
            index += (i + 1)
            summ += light[i]
            if index == summ:
                ins += 1
        return ins
```
