class Solution {
public:
    int hammingWeight(uint32_t n) {
        
        int count = 0;
        
        while(n>0){
            
            if(n%2 == 1)//如果是十进制字符串则应除以10
                count++;
            
            n = n/2;
        }
        
        return count;
    }
};
