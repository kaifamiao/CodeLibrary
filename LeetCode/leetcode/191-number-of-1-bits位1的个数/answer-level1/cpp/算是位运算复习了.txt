```
class Solution
{
public:
    int hammingWeight(uint32_t n)
    {
        uint32_t hole = 1;
        int count = 0;
        for (int i = 0; i < 32; i++)
        {
            if ((hole & n) == hole)//按位与运算符，这一位同为1则表达式结果为1。
                count++;
            hole <<= 1;//左移一位。
        }
        return count;
    }
};
```
