# 实现代码
```csharp
public IList<int> GrayCode(int n)
{
    IList<int> list = new List<int> { 0 };
    int len;
    for (int i = 0; i < n; i++)
    {
        len = list.Count;//记录之前元素数量
        for (int j = len - 1; j >= 0; j--)
        {
            list.Add((int)(list[j] + Math.Pow(2, i)));//对之前元素进行逆序复制并加上“1”
        }
    }
    return list;
}
```
