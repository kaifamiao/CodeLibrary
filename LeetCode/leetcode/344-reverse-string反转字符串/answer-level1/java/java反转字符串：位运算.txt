class Solution {
    public void reverseString(char[] s) {
        for (int i = 0; i < s.length / 2; i++) {
            int p = s.length - 1 - i;
            s[i] = (char) (s[i] ^ s[p]);
            s[p] = (char) (s[i] ^ s[p]);
            s[i] = (char) (s[i] ^ s[p]);
        }
    }
}
