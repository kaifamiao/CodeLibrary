class Solution {
    public boolean isNumber(String s) {
        if(s.charAt(s.length()-1)>'A'&&s.charAt(s.length()-1)<'z')return  false;
         try{
           if(Double.valueOf(s).getClass().equals(Double.class))
               return true;
           else return false;
       }catch (Exception e){
           return false;
       }
    }
}
