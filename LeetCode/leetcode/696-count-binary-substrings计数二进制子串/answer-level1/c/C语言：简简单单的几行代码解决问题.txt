由于0和1是分别组合在一起的，也就是说满足要求的字符串（不会存在0和1混合的情况如：0101  1001）
```
int countBinarySubstrings(char * s){
    int n = 0, pre = 0, curr = 1, len = strlen(s)-1;
    for (int i = 0; i < len; ++i) {
        if (s[i] == s[i+1]) ++curr;
        else               {pre = curr; curr = 1;}
        if (pre >= curr) ++n;
    }
    return n;
}
```
