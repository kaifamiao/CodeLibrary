```
int* sortedSquares(int* a, int n, int* returnSize){
	*returnSize=n;
	if(n==0) return a;
	a[0]=a[0]*a[0];
	int j;
	for(int i=1;i<n;i++){
		int temp=a[i]*a[i];
		for(j=i-1;j>=0;j--){
			if(temp<a[j]) a[j+1]=a[j];
			else break;
		}
		a[j+1]=temp;
	}
	return a;
}
```

```
	int temp=a[low],pivoty=a[low];
	while(low<high){
		while(a[high]>=pivoty&&low<high) high--;
```

```
void MergeSort(int *a,int low,int mid,int high){
	
	int *b=(int *)malloc(sizeof(int)*(high+1));
	for(int i=low;i<=high;i++){
		b[i]=a[i];
	}
	int i=low,j=mid+1,temp=low;
	while(i<=mid&&j<=high){
		if(b[i]<b[j]){
			a[temp++]=b[i++];
		}
		else a[temp++]=b[j++];
```

8、双指针
```
int* sortedSquares(int* a, int n, int* returnSize){
	*returnSize=n;
	if(n==0) return a;
	int p=0,q=-1;
	for(int i=0;i<n;i++){
		if(a[i]>=0) {
			p=i;
			q=i-1;
			break;
		}
	} 
	int k=0;
	int *res=(int *)malloc(sizeof(int)*n);
	while(p<n&&q>=0){
		if(a[p]*a[p]<a[q]*a[q]){
			res[k++]=a[p]*a[p];
			p++;
		}
		else{
			res[k++]=a[q]*a[q];
			q--;
		} 
	} 
	while(p<n) {
		res[k++]=a[p]*a[p];
		p++;
	}
	while(q>=0){
		res[k++]=a[q]*a[q];
		q--;
	}
	return res;
}
```
