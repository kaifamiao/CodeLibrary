### 解题思路
1，单调栈，记录左侧数组的下边界
2，如果当前数字大于栈顶元素时，说明栈顶元素开头的子数组只能延长到当前位置的前面一位
3，遍历完数组后，栈中剩余元素都可以延伸到数组最末尾
通过以上规则计算即可
时间复杂度 `O(n)`，空间复杂度`O(n)`

### 代码

```cpp
class Solution {
public:
    int validSubarrays(vector<int>& nums) {
        if (nums.size() < 2) return nums.size();
        int res = 0;
        stack<int> st;
        int N = nums.size();
        for (int i = 0; i < N; ++i) {
            while (!st.empty() && nums[st.top()] > nums[i]) {
                res += i - st.top();
                st.pop();
            }
            st.push(i);
        }
        while (!st.empty()) {
            res += N - st.top();
            st.pop();
        }
        return res;
    }
};
```
