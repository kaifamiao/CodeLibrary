```c
bool isSubsequence(char * s, char * t){
    // 如果*s和*t不为空，一直遍历
    while(*s && *t){
        // s序列的当前元素*s等于t序列的当前元素*t，说明匹配上了
        if(*s == *t){
            // s换成下一个元素
            s++;
        }
        // t换成下一个元素
        t++;
    }

    // 如果*s为空说明s是t的子序列
    // 如果*s不为空说明s不是t的子序列
    return *s == '\0';
}
```
