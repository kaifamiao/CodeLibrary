```python3
class Solution:
    def climbStairs(self, n: int) -> int:
        if n < 3:
            return n
        f = [0] * (n + 1)  # [0, n]
        f[1], f[2] = 1, 2
        for i in range(3, n + 1):
            f[i] = f[i - 1] + f[i - 2]
        return f[n]
```

```python3
class Solution:
    def climb(self,n, fn_2, fn_1, toStep):
        fn = fn_2 + fn_1
        if n == toStep:
            return fn
        return self.climb(n, fn_1, fn, toStep + 1)


    def climbStairs(self, n: int) -> int:
        if n < 3:
            return n
```
