**代码:**

```go
import (
    "math"
)
func climbStairs(n int) int {
    s5:=math.Sqrt(5);
    return int(math.Round((math.Pow((1+s5)/2 , float64(n+1)) -  math.Pow((1-s5)/2, float64(n+1))) /s5));
}
```
