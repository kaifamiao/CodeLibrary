题解
```
func judgeCircle(moves string) bool {
    up, left := 0, 0
    for _, v := range moves {
        if v == 85 {
            up ++
        } else if v == 68 {
            up--
        } else if v == 76 {
            left++
        } else {
            left--
        }
    }
    return up == 0 && left == 0
}
```
