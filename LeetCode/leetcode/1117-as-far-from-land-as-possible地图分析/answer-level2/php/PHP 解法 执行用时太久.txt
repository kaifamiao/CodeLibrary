class Solution {

    /**
     * @param Integer[][] $grids
     * @return Integer
     */
    function maxDistance($grids) {
        $dx = [0, 0, 1, -1];
        $dy = [1, -1, 0, 0];

        $queue = [];
        // grids length
        $len = count($grids);
        $item_len = count($grids[0]);
        // æŠŠæ‰€æœ‰é™†åœ°å…¥é˜Ÿåˆ—
        for($i = 0; $i < $len; ++$i) {
            for($j = 0; $j < $len; ++$j ) {
                if($grids[$i][$j] === 1) {
                    array_unshift($queue, [$i, $j]);
                }
            }
        }

        // ä»å„ä¸ªé™†åœ°å¼€å§‹ä¸€åœˆåœˆéå†æµ·æ´‹ï¼Œæœ€åéå†åˆ°çš„æµ·æ´‹å°±æ˜¯ç¦»é™†åœ°æœ€è¿œçš„æµ·æ´‹
        $hasOcean = false;
        $point = null;
        while(!empty($queue)) {
            $point = $queue[count($queue)-1];
            $x = $point[0];
            $y = $point[1];
            // var_dump($queue);
            // var_dump($x);
            // var_dump($y);
            array_pop($queue);
            // å–å‡ºé˜Ÿåˆ—çš„é™†åœ°å…ƒç´ ï¼Œå°†å…¶å››å‘¨(ä¸Šä¸‹å·¦å³)çš„æµ·æ´‹å…¥é˜Ÿåˆ—
            for($i = 0; $i < 4; ++$i) {
                $newX = $x + $dx[$i];
                $newY = $y + $dy[$i];
                if($newX < 0
                || $newX >= $len
                || $newY < 0
                || $newY >= $item_len
                || $grids[$newX][$newY] !== 0) {
                    continue;
                }
                 // è¿™é‡Œç›´æ¥ä¿®æ”¹åŸæ•°ç»„ï¼Œå°±ä¸éœ€è¦é¢å¤–çš„æ•°ç»„æ¥æ ‡å¿—æ˜¯å¦è®¿é—®
                $grids[$newX][$newY] = $grids[$x][$y] + 1;
                $hasOcean = true;
                array_unshift($queue, [$newX, $newY]);
            }
        }
        // æ²¡æœ‰é™†åœ°æˆ–è€…æ²¡æœ‰æµ·æ´‹ï¼Œç›´æ¥return -1
        if($point == null || !$hasOcean) {
            return -1;
        }

        // è¿”å›æœ€åä¸€æ¬¡=éå†åˆ°çš„æµ·æ´‹çš„è·ç¦»
        return $grids[$point[0]][$point[1]] - 1;
    }
