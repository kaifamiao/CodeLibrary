public class Land{
    int x;
    int y;
    Land(int x,int y) {
        this.x = x; 
        this.y = y;
    }
}
class Solution {
    private static int[] direct_x =new int[]{-1,+1,0,0};//ä¸Šä¸‹å·¦å³
    private static int[] direct_y =new int[]{0,0,-1,+1};
    public int maxDistance(int[][] grid) {
        int m = grid.length;//åœ°å›¾é«˜
        int n = grid[0].length;//åœ°å›¾é•¿
        Queue<Land> queue = new LinkedList<>();//é˜Ÿåˆ—
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                //è®°å½•é™†åœ°
                if(grid[i][j]==1){
                    queue.add(new Land(i,j));
                }
            }
        }
        if(queue.isEmpty()||queue.size()==m*n)return -1;
        //é˜Ÿåˆ—éç©ºå¾ªç¯
        Land land = null;
        while(!queue.isEmpty()){
            int size = queue.size();
            for(int i=0;i<size;i++){
                land = queue.poll();
                for(int d=0;d<4;d++){
                    int newX=land.x+direct_x[d];
                    int newY=land.y+direct_y[d];
                    //ä¸‹æ ‡è¶Šç•Œï¼Œéæµ·æ´‹åŒºåŸŸ
                    if(newX < 0 || newY < 0 || newX >= m || newY >= n || grid[newX][newY] != 0){
                        continue;
                    }
                    grid[newX][newY] = grid[land.x][land.y]+1;
                    queue.add(new Land(newX,newY));
                }
            }
        }
        return grid[land.x][land.y]-1;

    }
}
