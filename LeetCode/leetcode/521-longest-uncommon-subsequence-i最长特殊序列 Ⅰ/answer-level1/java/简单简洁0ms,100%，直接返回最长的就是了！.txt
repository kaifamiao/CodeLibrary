class Solution {
    public int findLUSlength(String a, String b) {
        if (a.equals(b))return -1;
        
        String LenStr = a.length()>b.length()?a:b;
        String ShoStr = a.length()<b.length()?a:b;
        
        return LenStr.length();

    }
}
