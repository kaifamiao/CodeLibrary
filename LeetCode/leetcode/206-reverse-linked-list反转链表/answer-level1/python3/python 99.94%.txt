# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def reverseList(self, head):
        if not head or head.next == None:
            return head
        left = head
        mid = head
        todo = head.next
        while todo:
            mid.next = todo.next
            todo.next = left
            left = todo
            todo = mid.next
        return left
