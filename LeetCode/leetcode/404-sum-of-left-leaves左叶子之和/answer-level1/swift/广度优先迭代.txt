/**
 * Definition for a binary tree node.

class Solution {
    func sumOfLeftLeaves(_ root: TreeNode?) -> Int {
        
        //åˆå§‹åŒ–ä¸€ä¸ªå…ƒç´ ç±»å‹ä¸ºå…ƒç»„çš„æ•°ç»„æ¨¡æ‹Ÿstack
        var stack = [(root,0)]
        //å®šä¹‰è¿”å›å€¼
        var result = 0
        
        
        //å¹¿åº¦ä¼˜å…ˆéå†ğŸŒ²ï¼Œåªè®°å½•å·¦å­çš„å€¼ï¼Œå½“æ¢ç´¢åˆ°å¶æ—¶å°†å·¦å¶çš„å€¼ç´¯åŠ åˆ°è¿”å›å€¼é‡Œ
        while let node = stack.last?.0, let value = stack.last?.1 {

            stack.removeLast()
            
            if node.left != nil {
                stack.append((node.left!, node.left!.val))
            }
            
            if node.right != nil {
                stack.append((node.right!, 0))
            }
            
            if node.left == nil && node.right == nil {
                result += value
            }
        }
        
        return result
        
    }
}
