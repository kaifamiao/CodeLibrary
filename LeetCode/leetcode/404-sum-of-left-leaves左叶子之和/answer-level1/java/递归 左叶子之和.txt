class Solution {  
    public int sumOfLeftLeaves(TreeNode root) {
        if(root==null){
            return 0;
        }
        int ans=0;
        TreeNode left=root.left;
        if(root.left!=null&&left.left==null&&left.right==null){
            ans+=root.left.val;
        }
        ans+=sumOfLeftLeaves(root.left)+sumOfLeftLeaves(root.right);
        return ans;
    }
}
