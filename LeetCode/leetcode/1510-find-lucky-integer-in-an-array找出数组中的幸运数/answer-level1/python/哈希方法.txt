class Solution:
    def findLucky(self, arr: List[int]) -> int:
        length = len(arr)
        nums = [0 for i in range(length + 1)]
        for num in arr:
            if num <= length and num >= 0:
                nums[num] += 1
        result = -1
        for i in range(length + 1):
            if i == nums[i] and nums[i] != 0:
                result = nums[i]
        return result
