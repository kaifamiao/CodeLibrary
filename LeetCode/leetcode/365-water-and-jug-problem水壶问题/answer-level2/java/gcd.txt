public boolean canMeasureWater(int x, int y, int z) {
    if (x + y < z) {
        return false;
    }

    if (x == 0 || y == 0) {
        return z == 0 || x + y == z;
    }

    return z % gcd(x, y) == 0;
}

private int gcd(int a, int b) {
    while (b != 0) {
        int temp = a % b;
        a = b;
        b = temp;
    }
    return a;
}
