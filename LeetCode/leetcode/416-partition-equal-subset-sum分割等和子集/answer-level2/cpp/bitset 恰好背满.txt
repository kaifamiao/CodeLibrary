class Solution {
public:
    bool canPartition(vector<int>& nums) {
        int ss = accumulate(nums.begin(), nums.end(), 0);
        if (ss % 2) {
            return false;
        }
        bitset<10005> bs(1);
        for (auto i: nums) {
            bs |= (bs << i);
            if (bs[ss / 2]) {
                return true;
            }
        }
        return false;
    }
};
