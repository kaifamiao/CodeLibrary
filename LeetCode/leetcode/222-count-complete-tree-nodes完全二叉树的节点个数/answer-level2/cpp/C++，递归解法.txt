class Solution {
public:
    int count = 0;
    int countNodes(TreeNode* root) {
        if(root == NULL){
            return 0;
        }
        countNodes(root->left);
        countNodes(root->right);
        if(root){
             count++;
        }
        return count;     
    }
};
