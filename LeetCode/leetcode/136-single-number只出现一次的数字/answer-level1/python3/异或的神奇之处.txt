class Solution:
    def singleNumber(self, nums: List[int]) -> int:
        if len(nums) == 0:
            return
        if len(nums) == 1:
            return nums[0]
        if len(nums) ==2:
            return
        val_in_nums = nums[0]
        for i in range(1,len(nums)):
            val_in_nums = val_in_nums ^ nums[i]
        return val_in_nums
