```java
public int singleNumber(int[] nums) {
		 int j;
		 for(int i=0;i<nums.length;i++) {
			 if(nums[i]==-1) continue;
			 int targer = nums[i];
			 for(j=i+1;j<nums.length;j++) {
				 if(nums[j]==targer) { nums[j]=-1;break;}
			 }
			 if(j==nums.length) return targer;
		 }
		 return -1;
	    }
```
