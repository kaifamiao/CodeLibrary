/**
 * @param {number} rowIndex
 * @return {number[]}
 */
var getRow = function(rowIndex) {
    let arr = [1];
    for(let i=1;i<=rowIndex;i++){
        arr.push([arr[i-1]*(rowIndex-i+1)/i]);
    }
    return arr;
};```
