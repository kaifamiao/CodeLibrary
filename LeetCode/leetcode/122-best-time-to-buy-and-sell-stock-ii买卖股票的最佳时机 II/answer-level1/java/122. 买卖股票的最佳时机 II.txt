```
//所有正的delta加起来即可
class Solution {
    public int maxProfit(int[] prices) {
        int ans, i;

        ans = 0;
        for(i = 1; i < prices.length; i++)
            if(prices[i] > prices[i-1])
                ans += prices[i] - prices[i-1];
        return ans;
    }
}
```
