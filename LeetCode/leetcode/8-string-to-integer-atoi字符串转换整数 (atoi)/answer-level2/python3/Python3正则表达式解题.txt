class Solution:
    def myAtoi(self, str: str) -> int:
        import re
        re_str = r"\s*[-|+]?\d+"
        result = re.match(re_str, str)
        try:
            result = int(result[0].strip(" "))
        except:
            return 0
        else:
            return max(min(result, 2**31 - 1), (-2)**31)
