class Solution {

    /**
     * @param String $J
     * @param String $S
     * @return Integer
     */
    function numJewelsInStones($J, $S) {
        $map = [];
        for ($i = 0; $i < strlen($J); $i++) {
            $map[] = $J[$i];
        }

        $count = 0;
        for ($j = 0; $j < strlen($S); $j++) {
            if (in_array($S[$j], $map)) $count++;
        }

        return $count;
    }
}
