以及第二种方法,先将字符串排序，再比较

```
class Solution {
public:
    bool isAnagram(string s, string t) {
        if(s.size()!=t.size())
        { 
            return false;
        }sort(s.begin(),s.end());
        sort(t.begin(),t.end());
        return s==t;
    }
};
```
