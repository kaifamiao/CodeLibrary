class Solution {
public:
    int lastStoneWeight(vector<int>& stones) {
        int zeros = stones.size();
        while(zeros > 1)
        {
            sort(stones.rbegin(), stones.rend());
            if(stones[1] == 0)
                break;
            stones[0]-=stones[1];
            stones[1]-=stones[1];
        }
        return stones[0];
    }
};
