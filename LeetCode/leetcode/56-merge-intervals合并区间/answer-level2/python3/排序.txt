class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
        intervals.sort(key = lambda x:x[0])
        res = []
        for temp in intervals:
            if not res or res[-1][1] < temp[0]:
                res.append(temp)
            else:
                res[-1][1] = max(res[-1][1], temp[1])
        return res
