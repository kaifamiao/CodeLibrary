var uniqueLetterString = function(S) {
   var sum = 0;
		for (var i = 0; i < S.length; ++i) {
			for (var j = i - 1; j > -1 && S[j] != S[i]; --j);
			for (var k = i + 1; k < S.length && S[k] != S[i]; ++k);
			sum += (i - j)*( k - i);
		}
		return sum%1000000007;
};
