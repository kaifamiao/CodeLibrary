class Solution(object):
    def isPowerOfFour(self, num):
        """
        :type num: int
        :rtype: bool
        """
        if num<0:
            return False
        num=bin(num)[2::]
        if num.count('1')==1 and len(num)%2!=0:
            return True
        return False
