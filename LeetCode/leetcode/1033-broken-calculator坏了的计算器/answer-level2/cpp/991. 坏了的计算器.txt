class Solution {
public:
    int brokenCalc(int X, int Y) {
        if(X>=Y)return X-Y;
        vector<int> q;
        int t=Y;
        while(t>X)
        {
            q.emplace_back(t);
            t=(t+1)/2;
        }
        int res=X-t;
        for(int i=q.size()-1;i>=0;--i)
        {
            res+=(t*2-q[i]+1);
            t=q[i];
        }
        return res;
    }
};
