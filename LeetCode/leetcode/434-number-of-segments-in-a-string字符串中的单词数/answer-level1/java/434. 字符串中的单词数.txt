public int countSegments(String s) {
    int res = 0;
    //注意java中的split函数在切分过程中会出现 ""
    String[] s_arr = s.split(" ");
    for(String s1 : s_arr){
        if(s1.length() != 0){
            res ++;
        }
    }

    return res;
}
