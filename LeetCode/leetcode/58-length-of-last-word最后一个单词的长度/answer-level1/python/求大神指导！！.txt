class Solution(object):
    def lengthOfLastWord(self, s):
        """
        :type s: str
        :rtype: int
        """
        if s=='':
            return 0
        ws=s.split(' ')
        i=len(ws)-1
        while i>=0:
            if ws[i]!=' ':
                return len(ws[i])
                break
            else:
                i=i-1
