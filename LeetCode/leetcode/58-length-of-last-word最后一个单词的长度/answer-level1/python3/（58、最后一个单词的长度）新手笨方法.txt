class Solution:
    def lengthOfLastWord(self, s: str) -> int:
        if len(s) == 0:
            return 0
        s = s.strip()  # 返回去除首尾空格符后的字符串
        j = 0
        for i in range(len(s)-1,-1,-1):
            if s[i] == ' ':
                break
            j += 1
        return j
