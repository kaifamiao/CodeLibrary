    public boolean canJump(int[] nums) {
        if (nums == null || nums.length == 0) {
            return false;
        }
        if (nums.length == 1) {
            return true;
        }
        int tmp = 0;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] > tmp) {
                tmp = 0;
            } else {
                tmp++;
            }
        }
        return tmp == 0;
    }
