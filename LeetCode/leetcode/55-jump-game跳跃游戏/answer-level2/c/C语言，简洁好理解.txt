    bool canJump(int* nums, int numsSize){
    int i,tmp;
    
    for(i=numsSize-2,tmp=numsSize-1;i>=0;i--){
        if(nums[i]>=(tmp-i)){          //如果i点能够跳到tmp，令i为tmp
            tmp=i;
        }
    }

    if(tmp==0) return 1;
    else return 0;
    }
