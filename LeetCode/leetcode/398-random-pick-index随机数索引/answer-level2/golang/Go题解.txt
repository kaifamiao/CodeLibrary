type Solution struct {
	arr []int
}


func Constructor(nums []int) Solution {
	return Solution{arr:nums}
}


func (this *Solution) Pick(target int) int {
	res:=-1
	temp:=1
	for i:=0;i<len(this.arr);i++ {
		if this.arr[i]==target {
			if res==-1 {
				res=i
			}else{
				temp++
				if rand.Intn(temp)==0{
					res=i
				}
			}
		}
	}
	return res
}

 * param_1 := obj.Pick(target);
