class Solution {
public:
    int search(vector<int>& nums, int target) {
        int low = 0;
        int high=nums.size()-1;
        while(low<high)
        {
            int mid=(low+high)>>1;
            if(nums[mid]<target)
            {
                low=mid+1;
            }
            else
            {
                high=mid;
            }
        }

        return (nums[low]==target)?low:-1;
    }
};
