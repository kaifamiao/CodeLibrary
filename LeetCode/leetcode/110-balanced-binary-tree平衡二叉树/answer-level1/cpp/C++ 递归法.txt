class Solution {
public:
    bool isbalance = true;
    bool isBalanced(TreeNode* root) {
        treeHeight(root);
        return isbalance;
    }
    
    int treeHeight(TreeNode* root){
        if(root == NULL)
            return 0;
        int l = treeHeight(root->left);
        int r = treeHeight(root->right);
        if(abs(l-r)>1)
            isbalance = false;
        return max(l,r)+1;
    }
};
