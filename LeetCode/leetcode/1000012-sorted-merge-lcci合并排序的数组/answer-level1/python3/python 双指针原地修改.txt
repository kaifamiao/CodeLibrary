class Solution:
    def merge(self, A: List[int], m: int, B: List[int], n: int) -> None:
        """
        Do not return anything, modify A in-place instead.
        """
        p = m+n-1
        pa = m-1
        pb = n-1
        while pa>=0 or pb>=0:
            if pa==-1:
                A[p] = B[pb]
                pb-=1
            elif pb==-1:
                A[p] = A[pa]
                pa-=1
            elif A[pa]>B[pb]:
                A[p] = A[pa]
                pa-=1
            else :
                A[p] = B[pb]
                pb-=1
            p-=1
