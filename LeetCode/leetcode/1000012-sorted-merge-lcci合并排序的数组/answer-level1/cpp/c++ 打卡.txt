class Solution {
public:
    void merge(vector<int>& A, int m, vector<int>& B, int n) {
        int i = 0, j = 0;
        int temp[m + n];
        int cur;
        while (i < m || j < n) {
            if (i == m)
                cur = B[j++];
            else if (j == n)
                cur = A[i++];
            else if (A[i] < B[j])
                cur = A[i++];
            else if(A[i] >= B[j])
                cur = B[j++];
            temp[i + j - 1] = cur;
        }
        for (int k = 0; k < (m + n); ++k)
            A[k] = temp[k];
    }
};

class Solution {
public:
    void merge(vector<int>& A, int m, vector<int>& B, int n) {
        while(m > 0 && n > 0){
            if(A[m - 1] > B[n - 1]){
                A[m + n - 1] = A[m - 1];
                m--;
            }
            else{
                A[m + n - 1] = B[n - 1];
                n--;
            }      
        }
        while(n > 0){
            A[n - 1] = B[n - 1];
            n--;
        }
    }
};
