3、ES6的扩展运算符
```
var merge = function(A, m, B, n) {
    A.splice(m , A.length , ...B)
    
    //对A进行排序
    A.sort(function(a,b){
        if(a>b){
            return 1
        }else{
            return -1
        }
    })
    return A
};
```
