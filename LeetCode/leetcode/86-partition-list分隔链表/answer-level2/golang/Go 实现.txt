 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func partition(head *ListNode, x int) *ListNode {
    if head == nil{
        return head
    }
    dummnyhead1 := &ListNode{
        Val:-1,
        Next:nil,
    }

            cur1 = cur1.Next

            cur = next

}
