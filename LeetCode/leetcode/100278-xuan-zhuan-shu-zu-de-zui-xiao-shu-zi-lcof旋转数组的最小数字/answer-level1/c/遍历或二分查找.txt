int minArray(int* numbers, int numbersSize){
    int ans = numbers[0];
    for (int i = 1; i < numbersSize; ++i) {
        if (numbers[i] < numbers[i - 1]) {
            ans = numbers[i];
            break;
        }
    }
    return ans;
}

int minArray(int* numbers, int numbersSize){
    int i = 0, j = numbersSize - 1;
    while (i < j) {
        int m = (i + j) / 2;
        if (numbers[m] < numbers[j]) {
            j = m;
        }
        else if (numbers[m] > numbers[j]) {
            i = m + 1;
        }
        else {
            j -= 1;
        }
    }
    return numbers[i];
}
