```
class Solution:
    def twoSum(self, numbers: List[int], target: int) -> List[int]:
        hashmap={}
        for ind,num in enumerate(numbers):
            hashmap[num] = ind
        for i in range(len(numbers)):
            j = hashmap.get(target-numbers[i])
            if j != None and i!=j:
                return [i+1,j+1]
```
