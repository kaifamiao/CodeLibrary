class Solution {
public:
    char findTheDifference(string s, string t) {
        int sum = 0;
        for (auto i : t)  //计算t串的ascii码总值
            sum += i;
        for (auto i : s)  //利用sum减去s串的ascii总值
            sum -= i;
        return sum;
    }
};
