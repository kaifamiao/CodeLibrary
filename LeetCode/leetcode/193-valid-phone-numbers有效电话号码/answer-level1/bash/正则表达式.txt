```sh
#扩展正则表达式  ()|{}?+ 全是特殊字符，如果需要使用普通字符，需要转义
grep -E '^([0-9]{3}-|\([0-9]{3}\) )[0-9]{3}-[0-9]{4}$' file.txt
sed --quiet -E '/^([0-9]{3}-|\([0-9]{3}\) )[0-9]{3}-[0-9]{4}$/p' file.txt
awk '/^([0-9]{3}-|\([0-9]{3}\) )[0-9]{3}-[0-9]{4}$/{printf $0"\n"}' file.txt
#基本正则表达式 ()|{}?+ 全是普通字符，如果需要使用特殊功能，需要转义
grep --basic-regexp '^\([0-9]\{3\}-\|([0-9]\{3\}) \)[0-9]\{3\}-[0-9]\{4\}$' file.txt
```
