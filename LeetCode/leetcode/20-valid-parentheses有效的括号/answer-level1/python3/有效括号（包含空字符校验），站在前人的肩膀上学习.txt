代码如下：

class Solution:
    def isValid(self, s: str) -> bool:
        dic1 = {'(':')','[':']','{':'}'}
        stack = []
        for i in s:
            # print(i)
            if i in dic1:
                stack.append(i)
            elif len(stack)==0  : 
                return False
            elif i==" ":
                continue
            elif dic1[stack[-1]]!=i:
                    return False
            else:
                stack.pop()
        return len(stack)==0
test = Solution()
