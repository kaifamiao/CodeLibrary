class Solution {
public:
    string removeDuplicates(string s, int k) {
        for (int i = 0; i < s.size(); i++)
        {
            if(s[i]!=s[i+1]) continue;

            int j = i+1;
            while(s[i]==s[j] && j<s.size() && j<i+k)
                j++;
            if((j - i) < k) continue;
            s.erase(s.begin()+i, s.begin()+j);
            i = i >= k ? i-k : -1;
        }
        return s; 
    }
};
