二分查找借用69题思路，先找出num向下的平方根，再判断其平方与num是否相等

```
class Solution:
    def isPerfectSquare(self, num: int) -> bool:
        if num == 1:
            return True
        pre, cur = 0, num
        while pre < cur:
            mid = (pre + cur) // 2
            if mid**2 <= num and (mid+1)**2 > num:
                break
            elif mid**2 > num:
                cur = mid
            else:
                pre = mid
        if mid**2 == num:
            return True
        else:
            return False 
```
