int Comp (const void *a, const void *b)
{
    return *(int*)b - *(int*)a;
}

int triangleNumber(int* nums, int numsSize){
    qsort(nums, numsSize, sizeof(int), Comp);

    int result = 0;

    for (int i = 0; i < numsSize; i++) {
        int j = i + 1;
        int k = numsSize - 1;

        while (j < k) {
            if (nums[j] + nums[k] > nums[i]) {
                result += k - j;
                j++;
            } else {
                k--;
            }
        }
    }

    return result;
}
