class Solution {
    public int[] divingBoard(int shorter, int longer, int k) {
        //kä¸ªæœ¨æ¿ ç”±shorterå’Œlongerç»„æˆ
        if(k==0){
            return new int[0];
        }
        if(k==1){
            return new int[]{shorter,longer};
        }
        int res=k*shorter;
        if(shorter==longer){
            return new int[]{res};
        }
        int[] nums=new int[k+1];
        for(int i=0;i<=k;i++){
            //k=3 3 0
            //i=1 2 1
            //i=2 1 2
            //i=3 0 3
            int tmp=shorter*(k-i)+longer*(i);
            nums[i]=tmp;
        }
       Arrays.sort(nums);
       return nums;
    }
}
