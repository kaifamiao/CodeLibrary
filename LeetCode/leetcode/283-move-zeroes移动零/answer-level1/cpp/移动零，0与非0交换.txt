设置两个指针，分别指向0和非0元素，交换，并移动指针。
程序如下：
```cpp
class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        int zeros=0;
        int number=0;
        while(number<nums.size()){
            if(nums[number]!=0){
                swap(nums[zeros],nums[number]);
                zeros++;
            }
            number++;
        }
    }
};
```
