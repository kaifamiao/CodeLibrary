# 思路
以非ascii字符作为分割符
# 代码
```
class Codec:

    def encode(self, strs):
        """Encodes a list of strings to a single string.
        
        :type strs: List[str]
        :rtype: str
        """
        res = ""
        for i in strs:
            res+=i
            res+="\n"
        return res

    def decode(self, s):
        """Decodes a single string to a list of strings.
        
        :type s: str
        :rtype: List[str]
        """
        return s.split("\n")[:-1]
```
