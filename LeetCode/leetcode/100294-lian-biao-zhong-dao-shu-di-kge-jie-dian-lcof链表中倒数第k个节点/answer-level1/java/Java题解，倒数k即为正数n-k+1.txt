/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public ListNode getKthFromEnd(ListNode head, int k) {
        int len=0;
        ListNode cur=head;
        while(cur!=null){
            len++;
            cur=cur.next;
        }
        int num = len - k + 1; //倒数第k个节点，即为正数第n-k+1个结点
        while(--num>0){
            head=head.next;
        }
        return head;
    }
}
