var getKthFromEnd = function(head, k) {

    let node = head;
    while (node) {
        ++length;
        node = node.next;

        return null;
    }

    node = head;
    let offset = length - k;
    for (let i = 0; i < offset; ++i) {
        node = node.next;
    }
    return node;
};
