class Solution {
public:
    int reverse(int x) {
        int c = 0, m = 1, out = 0;
        int f = 1;
        if (x < 0) {
            f = -1;
        }
        while(x!=0) {
            c = x % 10;
            if((out>214748364||out==214748364&&c>7)||(out<-214748364||out==-214748364&&c<-8)){
                return 0;
            }
            out *= 10;
            out += c;
            x /= 10;
        }
        return out;
        
    }
};
