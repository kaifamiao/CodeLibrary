class Solution(object):
    def findContentChildren(self, g, s):
        """
        :type g: List[int]
        :type s: List[int]
        :rtype: int
        """
        res = 0
        g.sort()
        s.sort()
        
        g_length = len(g)
        s_length = len(s)
        
        i = 0
        j = 0
        while i < g_length and j < s_length:
            if g[i] <= s[j]:
                # å¯ä»¥æ»¡è¶³èƒƒå£ï¼ŒæŠŠå°é¥¼å¹²å–‚ç»™å°æœ‹å‹
                res += 1
                i += 1
                j += 1
            else:
                # ä¸æ»¡è¶³èƒƒå£ï¼ŒæŸ¥çœ‹ä¸‹ä¸€å—å°é¥¼å¹²
                j += 1
                
        return res
