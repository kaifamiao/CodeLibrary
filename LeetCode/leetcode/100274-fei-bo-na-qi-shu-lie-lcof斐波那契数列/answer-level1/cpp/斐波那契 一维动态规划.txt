class Solution {
public:
    int fib(int n)
    {
        int a = 0, b = 1, t;
        if (n == 0) return 0;//0和1是已知出口
        if (n == 1) return 1;
        for (int i = 2; i <= n; i++)
        {
            t = a; //利用t来记住前者
            a = b; //前者=后者
            b = t + b; //后者=前者+“后者” 斐波那契基本运算规律
            if (b > 1000000007) b %= 1000000007;//题设要求
        }
        return b;
    }
};

//一维动态规划，对于一个递归问题，如果子问题对于解决总问题有帮助，则我们可以考虑利用数组来记住子问题的结果，从而减少重复计算的时间
