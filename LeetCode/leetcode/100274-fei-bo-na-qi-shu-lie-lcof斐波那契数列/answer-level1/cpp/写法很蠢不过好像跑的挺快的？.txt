class Solution {
public:
    int fib(int n) {

        int f[101];
        f[0]=0;
        f[1]=1;
        for(int i=2;i<n+1;i++)
        {f[i]=(f[i-1]+f[i-2])%1000000007;}
        if(f[n]==1000000008)
        return 1;
      
        return f[n];

    }
};
