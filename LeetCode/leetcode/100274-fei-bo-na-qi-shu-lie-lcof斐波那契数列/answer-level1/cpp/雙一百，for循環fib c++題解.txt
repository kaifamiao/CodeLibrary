### 解题思路
使用臨時變量保存結果，基礎條件下直接返回結果，否則進入循環，下一次的循環可以利用這一次的計算結果
注意結果要取模，代碼注釋解釋特定要點。
### 代码

```cpp
class Solution {
public:
    int fib(int n) {
        int a=1,b=2,c=2;//c第一次循環被賦值于b，為保證循環正確讓c初始值與b相同
        if(n==0) return 0;
        if(n<3) return 1;
        else if(n==3) return 2;
        for(int i=3;i<n;i++)//n大于3時進入這裏
        {
            b=c;//保存上一次較大的數至b，為計算新的c做準備
            c=(a+b)%1000000007;
            a=b;//捨棄上一次的較小的數，保存下一次要用的，這一輪的比較大的數b
        }
        return c;
    }
};
```
