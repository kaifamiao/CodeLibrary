class Solution:
    def fib(self, n: int) -> int:
        if n <= 0:
            return 0
        elif n == 1:
            return 1
        else:
            a , b , i = 0 , 1 , 1
            while i < n:
                a, b = b, a + b
                i += 1
            return b % 1000000007
