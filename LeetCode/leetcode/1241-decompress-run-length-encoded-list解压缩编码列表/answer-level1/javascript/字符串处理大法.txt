/**
 * @param {number[]} nums
 * @return {number[]}

var decompressRLElist = function(nums) {
  return nums.join(',').match(/\d+,\d+/g).map(item => {
    const [len,num] = item.split(',');
    return (num + ',').repeat(len - 0).replace(/,$/,'');
  }).reduce((sum,curr) => sum + ',' + curr).split(',');
};
