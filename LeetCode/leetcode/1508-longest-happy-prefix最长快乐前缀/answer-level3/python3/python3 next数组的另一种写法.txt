class Solution:
    def longestPrefix(self, s: str) -> str:
        if len(s) <= 1:return ""
        n = len(s)
        dp = [[0] * 26 for _ in range(n)]
        dp[0][ord(s[0])-97],shadow = 1,0
        for i in range(1,n):
            for j in range(0,26):
                dp[i][j] = dp[shadow][j]
            dp[i][ord(s[i])-97] = i + 1
            shadow = dp[shadow][ord(s[i])-97]
        return s[:shadow]
