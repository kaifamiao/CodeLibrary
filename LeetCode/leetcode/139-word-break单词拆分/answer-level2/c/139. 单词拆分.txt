bool IsInDict(char * s, int i, int j, char ** wordDict, int wordDictSize)
{
    char *p_dict = NULL;
    char *p_s = (char *)(s + i);
    int num = j - i;
    int k = 0;
    int m = 0;

    for (k=0; k<wordDictSize; k++) {
        p_dict = wordDict[k];
        if (p_s[0] != p_dict[0] || strlen(p_dict) != num) {
            continue;
        } else {
            for (m = 1; m < num; m++) {
                if (p_s[m] != p_dict[m]) {
                    break;
                }
            }
            if (m == num) {
                return true;
            }
        }
    }
    return false;
}

bool wordBreak(char * s, char ** wordDict, int wordDictSize){
    if (s == NULL || wordDict == NULL || wordDict < 1) return false;
    int i=0;

    int num = strlen(s);
    int *di = (int *)malloc(sizeof(int) * (num + 1));
    if ( di == NULL) return false;
    di[0] = true;
    for (i=1; i<=num; i++) {
        di[i] = false;
    }
    int j=0;
    for (i=0; i<num; i++) {
        for (j=i+1; j<num+1; j++) {
            if (di[i] && IsInDict(s, i, j, wordDict, wordDictSize)) {
                di[j] = true;
            }
        }
    }

    bool ret = di[num];
    free(di);
    return ret;
}
