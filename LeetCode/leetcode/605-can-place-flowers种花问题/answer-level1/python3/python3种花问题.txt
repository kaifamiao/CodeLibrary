```
class Solution:
    def canPlaceFlowers(self, flowerbed: List[int], n: int) -> bool:
        left=0
        right=0
        for i in range(len(flowerbed)):
            if left==0 and flowerbed[i]==0 and (i+1==len(flowerbed) or flowerbed[i+1]==0):
                flowerbed[i]=1
                n-=1
            left=flowerbed[i]
        return n<=0
```
