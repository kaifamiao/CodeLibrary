class Solution {
    public int titleToNumber(String s) {
        int len = s.length();
        if(len==0)return 0;
        int res = 0;
        for(int i= 0; i<len; i++){
            char c = s.charAt(i);
            int k = c-'A'+1;    
           // System.out.println(k);
            res = res*26+k;
        }
        return res;
    }
}
