class NumArray {
private:
    vector<int> numset={0};
public:
    NumArray(vector<int>& nums) {
        for(int i=0;i<nums.size();i++){
            numset.push_back(nums[i]+numset[i]);
        }
    }
    int sumRange(int i, int j) {
        return numset[j+1]-numset[i];
    }
};
