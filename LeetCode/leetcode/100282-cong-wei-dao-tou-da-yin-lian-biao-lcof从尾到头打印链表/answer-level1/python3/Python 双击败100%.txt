```python3
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def reversePrint(self, head: ListNode) -> List[int]:
        q=head
        dp=[]
        while q:
            dp.append(q.val)
            q=q.next
        return dp[::-1]
```
