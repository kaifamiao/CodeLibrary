/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> reversePrint(ListNode* head) {
        vector<int> ans;
        if(head==NULL) return ans;
        ListNode* p=head;
        while(p)
        {
            ans.push_back(p->val);
            p=p->next;
        }
        reverse(ans.begin(),ans.end());
        return ans;
    }
};
